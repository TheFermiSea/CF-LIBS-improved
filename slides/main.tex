% ----------------------------------------------------------
% SLIDE 3: What is CF-LIBS?
% ----------------------------------------------------------
\begin{frame}{Calibration-Free LIBS}
	\begin{block}{The idea: derive composition from plasma physics alone}
		No matrix-matched standards needed --- use first-principles equations.
	\end{block}

	\vspace{0.5em}
	\textbf{Three governing equations:}
	\begin{enumerate}
		\item \textbf{Boltzmann distribution} $\to$ excited-state populations from $T$
		\item \textbf{Saha equation} $\to$ ionization ratios from $T$ and $n_e$
		\item \textbf{Closure} $\to$ $\sum_i C_i = 1$ (all concentrations sum to unity)
	\end{enumerate}

	\vspace{0.5em}
	\alert{Problem:} Computationally intensive --- $10^4$--$10^5$ transitions,
	iterative self-consistency, Voigt profile evaluation for every line.
\end{frame}

% ----------------------------------------------------------
% SLIDE 4: Why GPU acceleration?
% ----------------------------------------------------------
\begin{frame}{Why GPU Acceleration?}
	\textbf{Real-time demands:}
	\begin{itemize}
		\item Laser repetition: 1--20 Hz, 10--100 spectra accumulated per analysis
		\item Need hundreds of spectra/second for online monitoring
		\item Manifold-based inference: $10^6$--$10^8$ synthetic spectra
	\end{itemize}

	\vspace{0.5em}
	\textbf{Prior GPU spectroscopy work:}
	\begin{itemize}
		\item \textbf{ExoJAX} (Kawahara+ 2022): $\sim$10k spectra/s on V100 --- exoplanet atmospheres
		\item \textbf{HELIOS-K} (Grimm+ 2021): CUDA C++ opacity calculator, high throughput
	\end{itemize}

	\vspace{0.3em}
	\alert{Gap:} No GPU-accelerated CF-LIBS existed.
	Neither handles the multi-element Saha--Boltzmann chain.
\end{frame}

% ----------------------------------------------------------
% SLIDE 7: Kernel 1 — Voigt profiles
% ----------------------------------------------------------
\begin{frame}{Kernel 1: GPU Voigt Profile}
	\textbf{Challenge:} Evaluate Faddeeva function $w(z)$ on GPU
	\begin{itemize}
		\item Standard Humlicek W4 algorithm uses conditional branches\\
		      $\to$ warp divergence, incompatible with \texttt{jit}/\texttt{grad}
	\end{itemize}

	\vspace{0.5em}
	\textbf{Solution:} Weideman $N\!=\!36$ rational approximation
	\begin{equation*}
		w(z) \approx \frac{1}{\sqrt{\pi}} \frac{L}{L - iz}
		  + \frac{2}{(L - iz)^2} \sum_{n=0}^{35} c_n Z^n
	\end{equation*}

	\begin{itemize}
		\item \alert{Branch-free} --- no conditionals, pure arithmetic
		\item 236 FLOPs per evaluation
		\item Max relative error $< 10^{-13}$ in float64
		\item Full $(N_\text{wl} \times N_\text{lines})$ matrix in a single XLA kernel
	\end{itemize}
\end{frame}

% ----------------------------------------------------------
% SLIDE 8: Kernel 2 — Boltzmann fitting: the problem
% ----------------------------------------------------------
\begin{frame}{Kernel 2: Vectorized Boltzmann Fitting}
	\textbf{Boltzmann plot} extracts temperature from line intensities:
	\begin{equation*}
		\underbrace{\ln\!\left(\frac{I_i \,\lambda_i}{g_{k,i}\, A_{ki,i}}\right)}_{y_i}
		= \underbrace{-\frac{1}{k_B T}}_{b}\; E_{k,i} + \text{const}
		\qquad\Longrightarrow\qquad
		T = -\frac{1}{b\, k_B}
	\end{equation*}

	\vspace{-0.2em}
	{\small\textbf{Fit $y = a + bx$ by weighted least squares} --- minimize
	$\chi^2 = \sum_i w_i(y_i - a - bx_i)^2$.
	Setting $\partial\chi^2\!/\partial a = 0$, $\partial\chi^2\!/\partial b = 0$ gives the
	\alert{$2\times 2$ normal equations}:}
	\begin{equation*}
		\underbrace{
		\begin{pmatrix}
			S_w    & S_{wx} \\
			S_{wx} & S_{wxx}
		\end{pmatrix}}_{\bm{X}^\top \bm{W} \bm{X}}
		\begin{pmatrix} a \\ b \end{pmatrix}
		=
		\underbrace{
		\begin{pmatrix} S_{wy} \\ S_{wxy} \end{pmatrix}}_{\bm{X}^\top \bm{W} \bm{y}}
	\end{equation*}

	\vspace{-0.5em}
	{\small with \alert{five scalar accumulators}:
	$S_w\!=\!\sum w_i$,\;
	$S_{wx}\!=\!\sum w_i x_i$,\;
	$S_{wy}\!=\!\sum w_i y_i$,\;
	$S_{wxx}\!=\!\sum w_i x_i^2$,\;
	$S_{wxy}\!=\!\sum w_i x_i y_i$}
\end{frame}

% ----------------------------------------------------------
% SLIDE 8b: Boltzmann fitting: the solution
% ----------------------------------------------------------
\begin{frame}{Five Accumulators $\to$ Closed-Form Solution}
	\textbf{Analytic inverse} via adjugate (Cramer's rule) --- \alert{no matrix factorization}:
	\begin{equation*}
		b = \frac{S_w\, S_{wxy} - S_{wx}\, S_{wy}}{\Delta}, \qquad
		a = \frac{S_{wxx}\, S_{wy} - S_{wx}\, S_{wxy}}{\Delta}, \qquad
		\Delta = S_w\, S_{wxx} - S_{wx}^2
	\end{equation*}

	\vspace{-0.3em}
	{\small\textbf{Uncertainty} from Gram covariance $(\bm{X}^\top\bm{W}\bm{X})^{-1}$:\quad
		$\sigma_b = \sqrt{S_w / \Delta}$,\quad
		$\sigma_T = T^2 k_B\, \sigma_b$}

	\vspace{0.2em}
	\begin{columns}[T]
		\begin{column}{0.48\textwidth}
			{\small\textbf{Why not QR/SVD?}}
			\begin{itemize}
				\item {\small $2{\times}2$ system: $\kappa \sim 10^2$--$10^3$}
				\item {\small Loses 2--3 digits $\to$ 12--13 correct in \texttt{float64}}
				\item {\small Physical uncertainties $\sim 1$--$10\%$}
				\item {\small QR adds $3\times$ cost, zero practical benefit}
			\end{itemize}
		\end{column}
		\begin{column}{0.48\textwidth}
			{\small\textbf{Why this matters for GPU:}}
			\begin{itemize}
				\item {\small \alert{Branch-free}: same path for every element}
				\item {\small No warp divergence (no pivoting)}
				\item {\small Only $10 N_\text{max} + 11$ FLOPs/element}
				\item {\small Degenerate: branchless \texttt{where} $\to$ $T\!=\!0$}
			\end{itemize}
		\end{column}
	\end{columns}
\end{frame}

% ----------------------------------------------------------
% SLIDE 8c: Pad-and-mask batching
% ----------------------------------------------------------
\begin{frame}{Pad-and-Mask: Ragged $\to$ Rectangular $\to$ GPU}
	{\small\textbf{Problem:} $D$ elements with \emph{different} line counts (Li: 3, Ca: 12, Fe: 47)}

	\vspace{0.1em}
	{\small\textbf{Solution:} Pad to $N_\text{max}$, mask out padding via weight-zeroing $\tilde{w}_{s,i} = w_{s,i} \cdot m_{s,i}$}
	\begin{center}
		\vspace{-0.3em}
		\begin{tikzpicture}[
			cell/.style={minimum width=0.5cm, minimum height=0.38cm, font=\tiny, anchor=center},

			elabel/.style={font=\tiny\bfseries, anchor=east}
		]

			\node[elabel] at (-0.2, -0.5) {Ca};

			% Li: 2 valid + 3 padded
			\foreach \i in {0,1} { \node[valid] at (0.5*\i+0.25, -1.0) {$w_{\pgfmathparse{int(\i+1)}\pgfmathresult}$}; }

		\end{tikzpicture}
		\vspace{-0.3em}
	\end{center}

	\begin{columns}[T]
		\begin{column}{0.50\textwidth}
			{\small\textbf{Implementation:}}
			\begin{itemize}
				\item {\small Arrays shape $(D, N_\text{max})$}
				\item {\small \texttt{jnp.sum(w\_masked * x, axis=1)}}
				\item {\small Single \texttt{@jit} kernel, \texttt{float64}}
			\end{itemize}
		\end{column}
		\begin{column}{0.46\textwidth}
			{\small\textbf{Cost analysis:}}
			\begin{itemize}
				\item {\small $10\, N_\text{max} + 11$ FLOPs/element}
				\item {\small $D\!=\!10$, $N_\text{max}\!=\!50$: \alert{5{,}110 FLOPs}}
				\item {\small Fits in one GPU warp}
			\end{itemize}
		\end{column}
	\end{columns}
\end{frame}

% ----------------------------------------------------------
% SLIDE 9: Kernel 3 — Anderson acceleration
% ----------------------------------------------------------
\begin{frame}{Kernel 3: Anderson-Accelerated Saha Solver}
	\textbf{Fixed-point problem:} $n_e = g(n_e)$ via charge neutrality

	\vspace{0.5em}
	\textbf{Simple Picard iteration:} $n_e^{(k+1)} = g(n_e^{(k)})$
	\begin{itemize}
		\item Converges, but rate depends on $|g'(n_e^*)|$
	\end{itemize}

	\vspace{0.3em}
	\textbf{Anderson acceleration:} store $m$ previous iterates, solve least-squares
	\begin{equation*}
		n_e^{(k+1)} = g(n_e^{(k)}) - (\Delta G_k + \Delta R_k)\, \bm{\gamma}_k^*
	\end{equation*}

	\begin{itemize}
		\item Optimal depth $m = 1$ at tolerance $10^{-6}$
		\item Implemented via \texttt{lax.scan} --- static computation graph
		\item Reduces to Picard when $m = 0$ (zero-cost fallback)
	\end{itemize}
\end{frame}

% ----------------------------------------------------------
% SLIDE 10: Kernel 4 — Softmax closure
% ----------------------------------------------------------
\begin{frame}{Kernel 4: Softmax Compositional Closure}
	\textbf{Constraint:} $\sum_{i=1}^D C_i = 1$, \quad $C_i > 0$

	\vspace{0.5em}
	\textbf{Traditional approach:} Normalize after each iteration\\
	$\to$ breaks gradient flow, numerically fragile at simplex boundary

	\vspace{0.5em}
	\textbf{Softmax reparameterization:}
	\begin{equation*}
		C_i = \frac{\exp(\theta_i)}{\sum_{j} \exp(\theta_j)}
		\qquad\text{(log-sum-exp trick for stability)}
	\end{equation*}

	\begin{itemize}
		\item Constraint satisfied \alert{exactly by construction}
		\item Strict positivity for all finite $\bm{\theta}$
		\item Exact Jacobian: $J_{ij} = C_i(\delta_{ij} - C_j)$
		\item Enables gradient-based joint optimization of $T$, $n_e$, $\bm{C}$
	\end{itemize}

	\vspace{0.3em}
	Accuracy: max deviation $4.44 \times 10^{-16}$ (machine epsilon)
\end{frame}

% ----------------------------------------------------------
% SLIDE 11: Kernel 5 — Batch forward model
% ----------------------------------------------------------
\begin{frame}{Kernel 5: Batch Forward Model}
	{\small\textbf{Forward model:} $(T, n_e, \bm{C}) \to S(\lambda)$ via
	Saha $\to$ Boltzmann $\to$ emissivity $\to$ Voigt broadening}

	\vspace{0.2em}
	\textbf{Batch via \texttt{vmap}:}
	\begin{equation*}
		\texttt{batch\_spectrum} = \texttt{jit}(\texttt{vmap}(\texttt{single\_spectrum}))
	\end{equation*}

	\vspace{-0.3em}
	\begin{itemize}
		\item {\small Wavelength grid and atomic data \alert{shared} (broadcast)}
		\item {\small $(T, n_e, \bm{C})$ \alert{mapped} over batch dimension}
		\item {\small Profile matrix: $(B \times N_\text{wl} \times N_\text{lines})$ --- dominant memory cost}
		\item {\small \texttt{vmap} is a semantic transform: \alert{bit-identical} to sequential}
	\end{itemize}
\end{frame}

% SLIDE 23: Limitations
% ----------------------------------------------------------
\begin{frame}{Limitations and Caveats}
	\begin{itemize}
		\item {\small\textbf{Hardware-specific:} All numbers from Tesla V100S ---
		      consumer GPUs ($1/32$ float64 rate) would see lower speedups}
		\item {\small\textbf{LTE assumption:} Single-zone, optically thin, LTE only ---
		      non-LTE and self-absorption not yet GPU-accelerated}
		\item {\small\textbf{JIT overhead:} 30--120~s first call,
		      amortized over subsequent calls ($<$0.01\% for manifolds)}
		\item {\small\textbf{Float64 required:} Weideman coefficients span 14 orders of magnitude ---
		      float32 would double throughput but $\sim$0.1--1\% errors}
		\item {\small\textbf{GPU FAISS:} Not benchmarked (null results on test system)}
	\end{itemize}
\end{frame}

\begin{frame}{Backup: Key Mathematical Details}
	\textbf{Voigt --- Weideman $N\!=\!36$:}
	\begin{itemize}
		\item M\"obius transform: $Z = (L + iz)/(L - iz)$, $L \approx 5.0454$
		\item Degree-35 polynomial in $Z$, 236 FLOPs
		\item Branch-free $\to$ no warp divergence, compatible with \texttt{grad}
	\end{itemize}

	\vspace{0.5em}
	\textbf{Boltzmann --- Closed-form WLS:}
	\begin{equation*}
		b = \frac{S_w S_{wxy} - S_{wx} S_{wy}}{S_w S_{wxx} - S_{wx}^2}, \qquad
		T = \frac{-1}{b \cdot k_B}
	\end{equation*}

	\vspace{0.3em}
	\textbf{Softmax Jacobian:}
	\begin{equation*}
		J_{ij} = C_i(\delta_{ij} - C_j) \qquad \text{(rank $D\!-\!1$, $\kappa = 1$ at uniform)}
	\end{equation*}
\end{frame}
